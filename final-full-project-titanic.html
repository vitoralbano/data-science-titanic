def create_figure(title):
    """
        Criar uma figure com titulo posicionado
        
        Args:
            title (String): Valor a ser preenchido no ttulo do grfico 
            
        Return:
            (plt.Figure) Retorna uma Figure para adicionar grficos (Plots)
    """
    fig = plt.figure(title)
    fig.suptitle(title, fontsize='x-large', horizontalalignment='center', verticalalignment='bottom')
    plt.subplots_adjust(top=.9)
    plt.tight_layout()
    fig.set_figwidth(20)
    
    return fig

def get_survivors():
    """
        Exibir tabela com contagem de sobreviventes e proporo
        
        Return (dataframe)
    """
    # Definio do dataframe de propores
    df_survivors = df['sobreviveu'].value_counts().reset_index()
    df_survivors.rename(columns={'index': 'sobreviveu', 'sobreviveu': 'total'}, inplace=True)
    df_survivors['proporcao'] = round(df_survivors['total'].apply(lambda value: value / df_survivors['total'].sum()) * 100, 2)
    

    return df_survivors

def survivors_pie():
    df_survivors = get_survivors()
    fig = create_figure('')
    
    # pie plot
    ax_pie = fig.add_subplot(121)
    
    values = df_survivors['proporcao'] * 100
    explode = np.full(df_survivors.shape[0], 0.05)
    labels = df_survivors['sobreviveu'].replace(['Sim','No'],['Sobreviveu', 'No sobreviveu'])

    ax_pie.axis('equal')
    ax_pie.pie(
        values, 
        explode=explode, 
        labels=labels, 
        autopct='%1.1f%%', 
        shadow=True,
        startangle=90,
    );
    
    ax_pie.set_title('SOBREVIVENTES')
    
    # table plot
    ax_table = fig.add_subplot(122)
    table = ax_table.table(
        cellText = df_survivors.values,
        colLabels = ['SOBREVIVEU', 'TOTAL', 'PROPORO'],
        loc='center',        
    )
    ax_table.axis('off')
    ax_table.set_title('TABELA DE SOBREVIVENTES')
    table.auto_set_font_size(False)
    table.set_fontsize(13)

def most_survivors():
    df_surv = df.groupby(['classe','sexo'])['sobreviveu'].value_counts(normalize=True).to_frame()
    df_surv.columns = df_surv.columns.get_level_values(0)
    df_surv.rename(columns={'sobreviveu': 'percentual'}, inplace=True)
    df_surv['percentual'] = round(df_surv['percentual'] * 100, 2)
    df_surv.reset_index(inplace=True)
    
    ax = sns.factorplot(
        data = df_surv.query('sobreviveu == "Sim"'),
        x = 'classe',
        y = 'percentual',
        hue = 'sexo',
        kind = 'bar',
        palette = 'Blues_d',
    );
    
    plt.title('SOBREVIVENTES: CLASSIFICADOS POR SEXO E CLASSE')
    plt.ylabel('Percentual')
    plt.xlabel('Classe')
    plt.tight_layout()

Conclusão¶

Ao analisar a amostra de dados de passageiros do Titanic, foi possvel chegar as seguintes concluses.

  • A primeira classe apresentou representou o maior percentual de sobreviventes, com 62,6%, contra 47,2% da segunda e 24,2% da terceira. Tal diferena pode apontar que houve alguma facilidade ou favorecimento no acesso aos botes salva vidas para os passageiros da primeira classe.
  • Dentre os sobreviventes, 68% so mulheres e 32% homens
  • O grupo de classe e sexo que apresentou maior taxa de sobreviventes foi o feminino da primeira classe, com a relao de 96,7%
  • 59,0% das crianas a bordo foram salvas
  • A idade mdia entre os sobreviventes era de 28,4 anos.
  • O embarque de passageiros de Southampton representou 72,4%

Limitações¶

Inicialmente ao avaliar o conjunto de dados foi detectado que haviam campos como idade, embarcou e cabine, no preenchidos e que potencialmente prejudicariam a anlise. Assim, foram feitos alguns ajustes como:

  • Campos de idade com valor nulo foram preenchidos com a mdia
  • Foi efetuada uma breve anlise sobre os passageiros que apresentavam o campo 'embarcou' como nulo e se haviam sobrevivido, assim, dado que o retorno foi positivo, estes foram removidos do conjunto de dados considerando que estes no embarcaram.

Valores abreviados em local de embarque (embarcou) foram substitudos pelo nome do local sem abreviao.

Foram aplicadas tradues nos nomes das colunas e nos valores dos campos sobreviveu e sexo para auxiliar na criao de legendas dos grficos.

Foi criado um campo de categorizao da idade dos passageiros com o objetivo de mapea-los facilmente e levantar informaes a respeito de cada grupo.
